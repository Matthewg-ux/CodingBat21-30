function countEvens(arr){
    var even = 0;
    for(var i = 0; i < arr.length; i++){
        if(arr[i] % 2 == 0){
            even += 1;
        }
    }
    return even;

}

function countHi(str){

    var hi = 0;
    for(var i = 0; i < str.length; i++){
        if(str.substring(i, i+2) == "hi"){
            hi += 1;
        }
    }
    return hi;

}

function countTriple(str) {

    var end = -1;
    for(var i=0; i <= str.length;i++){
        if(str.substring(i,i+1) == str.substring(i+1,i+2) && str.substring(i+1,i+2) == str.substring(i+2,i+3)){
            end = end+1;
        }
    }
    return end;

}

function sameEnds(str) {

    var end = "";

    for(var i = 0; i < str.length; i++){
        if (str.substring(0, i) == str.substring(str.length - i, str.length) && i <= str.length-i) {
            end = str.substring(0, i);
        }
    }

    return end;
}
